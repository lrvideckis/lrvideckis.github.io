const int MAXN = 500'000;
int n;
long long t[4*MAXN];

void build(int a[], int tl=0, int tr=n, int v=1) {
    if (tr - tl == 1) {
        t[v] = a[tl];
    } else {
        int tm = (tl+tr)/2;
        build(a, tl, tm, 2*v);
        build(a, tm, tr, 2*v+1);
        t[v] = t[2*v] + t[2*v+1];
    }
}

long long query(int l, int r, int tl=0, int tr=n, int v=1) {
    if (r <= tl || tr <= l) return 0LL;
    if (l <= tl && tr <= r) return t[v];
    int tm = (tl+tr)/2;
    return query(l, r, tl, tm, 2*v)
         + query(l, r, tm, tr, 2*v+1);
}

void update(int pos, int new_val, int tl=0, int tr=n, int v=1) {
    if (tr - tl == 1) {
        t[v] += new_val;
    } else {
        int tm = (tl+tr)/2;
        if (pos < tm)
            update(pos, new_val, tl, tm, 2*v);
        else
            update(pos, new_val, tm, tr, 2*v+1);
        t[v] = t[2*v] + t[2*v+1];
    }
}
